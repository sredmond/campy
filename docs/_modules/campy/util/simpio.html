[docs]def get_yes_or_no(prompt=GETYESORNO_DEFAULT_PROMPT, reprompt=GETYESORNO_DEFAULT_REPROMPT, default=None):
    result = get_fn_cond(
        lambda line: line[0].upper() if line else default,
        lambda val: val in ['Y', 'N'],
        prompt, reprompt
    )
    return result == 'Y'




[docs]def get_fn_cond(fn, pred, prompt=DEFAULT_PROMPT, reprompt=DEFAULT_REPROMPT):
    prompt = append_space(prompt)
    while True:
        line = input(prompt)
        try:
            out = fn(line)
        except Exception:
            pass
        else:
            if pred(out):
                return out
        if reprompt:
            print(append_space(reprompt))

[docs]def append_space(prompt):
    """Adds a space to the end of the given string if none is present."""
    if not prompt.endswith(' '):
        return prompt + ' '
    return prompt
